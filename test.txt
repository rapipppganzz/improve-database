local GIST_ID = "e8c666a5a6d56b61695305d700a11d4c" -- ID Gist yang sudah Anda buat

local FILE_NAME_IN_GIST = "autowalk_paths.json"

local BASE_URL = "https://api.github.com/gists/" .. GIST_ID
local HEADERS = {
    ["Authorization"] = "token " .. GITHUB_TOKEN,
    ["Content-Type"] = "application/json",
    ["Accept"] = "application/vnd.github.v3+json"

-- Fungsi untuk mendapatkan nama map saat ini
local function GetCurrentMapName()
    return game.PlaceId .. "_" .. (Workspace:FindFirstChild("Map") and Workspace.Map.Name or "DefaultMap")
end

-- Fungsi internal untuk mengambil SELURUH konten Gist (sebagai tabel)
local function FetchAllData()
    if GIST_ID == "GANTI_DENGAN_GIST_ID_ANDA" or GITHUB_TOKEN == "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" then
        warn("GITHUB TOKEN/ID BELUM DIISI. Penyimpanan data Gagal.")
        return {}
    end
    
    local success, response = pcall(function()
        return HttpService:RequestAsync({
            Url = BASE_URL,
            Method = "GET",
            Headers = HEADERS
        })
    end)

    if success and response.Success and response.StatusCode == 200 then
        local data = HttpService:JSONDecode(response.Body)
        local fileInfo = data.files[FILE_NAME_IN_GIST]
        
        if fileInfo and fileInfo.content then
            local successDecode, savedTable = pcall(function()
                return HttpService:JSONDecode(fileInfo.content)
            end)
            return successDecode and savedTable or {}
        end
    end
    return {}
end

-- Fungsi untuk save recording ke database (GIST)
local function SaveRecording(path)
    local mapName = GetCurrentMapName()
    local existingData = FetchAllData() -- Ambil data lama dari Gist
    
    existingData[mapName] = path -- Update data untuk map saat ini

    local newFileContent = HttpService:JSONEncode(existingData)
    local body = {
        description = "Update RapipppXD Autowalk Data",
        files = {
            [FILE_NAME_IN_GIST] = {
                content = newFileContent
            }
        }
    }
    
    local success, response = pcall(function()
        return HttpService:RequestAsync({
            Url = BASE_URL,
            Method = "PATCH", -- Gunakan PATCH untuk update konten file
            Headers = HEADERS,
            Body = HttpService:JSONEncode(body)
        })
    end)

    if success and response.Success and (response.StatusCode == 200 or response.StatusCode == 201) then
        StarterGui:SetCore("SendNotification", {
            Title = "Recording Saved (GitHub)",
            Text = "Successfully saved to Gist.",
            Duration = 4
        })
        return true
    else
        warn("Failed to write Gist data. Status:", response and response.StatusCode)
        StarterGui:SetCore("SendNotification", {
            Title = "SAVE ERROR",
            Text = "Failed to save to GitHub Gist! Cek GIST ID/Token.",
            Duration = 5
        })
        return false
    end
end

-- Fungsi untuk load recording dari database (GIST)
local function LoadRecording()
    local mapName = GetCurrentMapName()
    local allData = FetchAllData()
    return allData[mapName] or {}
end

local function ForcePart(v)
    if v:IsA("Part") and not v.Anchored and not v.Parent:FindFirstChild("Humanoid") and not v.Parent:FindFirstChild("Head") and v.Name ~= "Handle" then
        for _, x in next, v:GetChildren() do
            if x:IsA("BodyAngularVelocity") or x:IsA("BodyForce") or x:IsA("BodyGyro") or x:IsA("BodyPosition") or x:IsA("BodyThrust") or x:IsA("BodyVelocity") or x:IsA("RocketPropulsion") then
                x:Destroy()
            end
        end
        if v:FindFirstChild("Attachment") then
            v:FindFirstChild("Attachment"):Destroy()
        end
        if v:FindFirstChild("AlignPosition") then
            v:FindFirstChild("AlignPosition"):Destroy()
        end
        if v:FindFirstChild("Torque") then
            v:FindFirstChild("Torque"):Destroy()
        end
        v.CanCollide = false
        local Torque = Instance.new("Torque", v)
        Torque.Torque = Vector3.new(100000, 100000, 100000)
        local AlignPosition = Instance.new("AlignPosition", v)
        local Attachment2 = Instance.new("Attachment", v)
        Torque.Attachment0 = Attachment2
        AlignPosition.MaxForce = 9999999999999999
        AlignPosition.MaxVelocity = math.huge
        AlignPosition.Responsiveness = 200
        AlignPosition.Attachment0 = Attachment2
        AlignPosition.Attachment1 = Attachment1
    end
end

local function PlayRecording()
    if #recordedPath == 0 then
        StarterGui:SetCore("SendNotification", {
            Title = "No Recording",
            Text = "Please record or load a path first!",
            Duration = 3
        })
        return
    end
    
    if isRecording then
        StopRecording()
    end
    
    isPlaying = true
    PlayButton.Text = "‚è∏ Stop"
    PlayButton.BackgroundColor3 = Color3.fromRGB(200, 100, 0)
    playSound("12221967")
    
    local currentIndex = 1
    
    playbackConnection = RunService.Heartbeat:Connect(function()
        if not isPlaying then
            if playbackConnection then
                playbackConnection:Disconnect()
                playbackConnection = nil
            end
            return
        end
        
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Humanoid") then
            local hrp = char.HumanoidRootPart
            local humanoid = char.Humanoid
            
            if currentIndex <= #recordedPath then
                local targetData = recordedPath[currentIndex]
                local distance = (hrp.Position - targetData.Position).Magnitude
                
                if distance < 3 then
                    currentIndex = currentIndex + 1
                else
                    humanoid:MoveTo(targetData.Position)
                end
            else
                -- Loop kembali ke awal
                currentIndex = 1
            end
        end
    end)
end

-- Ring Parts Logic
local radius = 50
local height = 100
local rotationSpeed = 0.5
local attractionStrength = 1000
local ringPartsEnabled = false

local function RetainPart(Part)
    if Part:IsA("BasePart") and not Part.Anchored and Part:IsDescendantOf(workspace) then
        if Part.Parent == LocalPlayer.Character or Part:IsDescendantOf(LocalPlayer.Character) then
            return false
        end
        Part.CustomPhysicalProperties = PhysicalProperties.new(0, 0, 0, 0, 0)
        Part.CanCollide = false
        return true
    end
    return false
end

local parts = {}
local function addPart(part)
    if RetainPart(part) then
        if not table.find(parts, part) then
            table.insert(parts, part)
        end
    end
end

local function removePart(part)
    local index = table.find(parts, part)
    if index then
        table.remove(parts, index)
    end
end

for _, part in pairs(workspace:GetDescendants()) do
    addPart(part)
end

workspace.DescendantAdded:Connect(addPart)
workspace.DescendantRemoving:Connect(removePart)

RunService.Heartbeat:Connect(function()
    if not ringPartsEnabled then return end
    
    local humanoidRootPart = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if humanoidRootPart then
        local tornadoCenter = humanoidRootPart.Position
        for _, part in pairs(parts) do
            if part.Parent and not part.Anchored then
                local pos = part.Position
                local distance = (Vector3.new(pos.X, tornadoCenter.Y, pos.Z) - tornadoCenter).Magnitude
                local angle = math.atan2(pos.Z - tornadoCenter.Z, pos.X - tornadoCenter.X)
                local newAngle = angle + math.rad(rotationSpeed)
                local targetPos = Vector3.new(
                    tornadoCenter.X + math.cos(newAngle) * math.min(radius, distance),
                    tornadoCenter.Y + (height * (math.abs(math.sin((pos.Y - tornadoCenter.Y) / height)))),
                    tornadoCenter.Z + math.sin(newAngle) * math.min(radius, distance)
                )
                local directionToTarget = (targetPos - part.Position).unit
                part.Velocity = directionToTarget * attractionStrength
            end
        end
    end
end)

-- Make GUI Resizable (Logic Baru)
local resizing = false
local resizeStartInput
local resizeStartSize
local minimumSize = Vector2.new(280, 250) -- Ukuran minimum agar GUI tidak terlalu kecil

local function updateResize(input)
    local delta = input.Position - resizeStartInput.Position
    local newSize = resizeStartSize + delta
    
    -- Pastikan ukuran tidak kurang dari minimum
    newSize = Vector2.new(math.max(minimumSize.X, newSize.X), math.max(minimumSize.Y, newSize.Y))
    
    -- Ubah ukuran MainFrame
    MainFrame.Size = UDim2.new(0, newSize.X, 0, newSize.Y)
    
    -- Atur ulang ukuran ScrollFrame agar konten tetap memenuhi bingkai
    ScrollFrame.Size = UDim2.new(1, -10, 1, -55) 
end

ResizeHandle.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        resizing = true
        resizeStartInput = input
        resizeStartSize = Vector2.new(MainFrame.Size.X.Offset, MainFrame.Size.Y.Offset)
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                resizing = false
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if resizing then
        updateResize(input)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if resizing and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
        resizing = false
        -- Perbarui originalSize agar fitur minimize/maximize menggunakan ukuran terbaru
        originalSize = MainFrame.Size 
    end
end)
