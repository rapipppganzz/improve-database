local Players = game:GetService("Players")

local TweenService = game:GetService("TweenService")
local player = Players.LocalPlayer

-- Stealth Configuration
local CONFIG = {
    -- Use Roblox-approved assets
    JumpscareSound = "rbxassetid://9111268332", -- Eerie ambient sound
    JumpscareImage = "rbxassetid://2768370884", -- Red alert image
    
    -- Subtle effect settings
    EffectDuration = 10, -- Shorter duration to avoid detection
    SkyTransitionTime = 3,
    FadeOutTime = 2
}

-- Stealth Variables
local isActive = false
local originalLighting = {}
local effectConnections = {}

-- Save original lighting settings
local function saveOriginalLighting()
    originalLighting = {
        Brightness = Lighting.Brightness,
        OutdoorAmbient = Lighting.OutdoorAmbient,
        FogColor = Lighting.FogColor,
        FogEnd = Lighting.FogEnd,
        Ambient = Lighting.Ambient
    }
end

-- Stealth lighting changes that are less detectable
local function applyStealthLighting()
    saveOriginalLighting()
    
    -- Gradual transition instead of instant change
    local tweenInfo = TweenInfo.new(CONFIG.SkyTransitionTime, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    
    local brightnessTween = TweenService:Create(Lighting, tweenInfo, {Brightness = 0.5})
    local ambientTween = TweenService:Create(Lighting, tweenInfo, {OutdoorAmbient = Color3.fromRGB(100, 0, 0)})
    local fogTween = TweenService:Create(Lighting, tweenInfo, {
        FogColor = Color3.fromRGB(80, 0, 0),
        FogEnd = 500
    })
    
    brightnessTween:Play()
    ambientTween:Play()
    fogTween:Play()
    
    -- Change skybox subtly
    local sky = Lighting:FindFirstChildOfClass("Sky")
    if sky then
        -- Use existing Roblox skyboxes that are red-themed
        sky.SkyboxBk = "rbxassetid://264909758" -- Red nebula
        sky.SkyboxDn = "rbxassetid://264909758"
        sky.SkyboxFt = "rbxassetid://264909758"
        sky.SkyboxLf = "rbxassetid://264909758"
        sky.SkyboxRt = "rbxassetid://264909758"
        sky.SkyboxUp = "rbxassetid://264909758"
    end
end

-- Create global sound that all players can hear (legitimate method)
local function createGlobalSound()
    local sound = Instance.new("Sound")
    sound.Name = "AmbientEffectSound"
    sound.SoundId = CONFIG.JumpscareSound
    sound.Volume = 0.3 -- Lower volume to be less intrusive
    sound.Looped = false
    sound.Parent = workspace
    
    -- Play sound at every player's position
    for _, otherPlayer in pairs(Players:GetPlayers()) do
        if otherPlayer.Character and otherPlayer.Character:FindFirstChild("Head") then
            local soundClone = sound:Clone()
            soundClone.Parent = otherPlayer.Character.Head
            soundClone:Play()
            
            -- Auto cleanup
            game:GetService("Debris"):AddItem(soundClone, 10)
        end
    end
    
    sound:Play()
    game:GetService("Debris"):AddItem(sound, 10)
end

-- Create screen effect for all players using legitimate methods
local function createScreenEffects()
    for _, otherPlayer in pairs(Players:GetPlayers()) do
        -- Use coroutine to avoid detection
        coroutine.wrap(function()
            local gui = Instance.new("ScreenGui")
            gui.Name = "VisualEffect"
            gui.ResetOnSpawn = false
            gui.IgnoreGuiInset = true
            
            local success = pcall(function()
                gui.Parent = otherPlayer:WaitForChild("PlayerGui")
            end)
            
            if success then
                -- Create red overlay (subtle)
                local frame = Instance.new("Frame")
                frame.Size = UDim2.new(1, 0, 1, 0)
                frame.BackgroundColor3 = Color3.fromRGB(80, 0, 0)
                frame.BackgroundTransparency = 0.7
                frame.BorderSizePixel = 0
                frame.Parent = gui
                
                -- Add jumpscare image (brief)
                local image = Instance.new("ImageLabel")
                image.Size = UDim2.new(0.5, 0, 0.5, 0)
                image.Position = UDim2.new(0.25, 0, 0.25, 0)
                image.BackgroundTransparency = 1
                image.Image = CONFIG.JumpscareImage
                image.ImageTransparency = 1
                image.Parent = gui
                
                -- Animate effects
                local fadeIn = TweenService:Create(frame, TweenInfo.new(1), {BackgroundTransparency = 0.3})
                local imageFadeIn = TweenService:Create(image, TweenInfo.new(0.5), {ImageTransparency = 0})
                local imageFadeOut = TweenService:Create(image, TweenInfo.new(0.5), {ImageTransparency = 1})
                local fadeOut = TweenService:Create(frame, TweenInfo.new(CONFIG.FadeOutTime), {BackgroundTransparency = 1})
                
                -- Sequence animations
                fadeIn:Play()
                wait(1)
                imageFadeIn:Play()
                wait(1)
                imageFadeOut:Play()
                wait(2)
                fadeOut:Play()
                
                -- Cleanup
                wait(CONFIG.FadeOutTime)
                gui:Destroy()
            end
        end)()
    end
end

-- Create particle effects in workspace (visible to all, but legitimate)
local function createParticleEffects()
    local part = Instance.new("Part")
    part.Name = "GlobalEffect"
    part.Size = Vector3.new(100, 100, 100)
    part.Position = Vector3.new(0, 50, 0)
    part.Anchored = true
    part.CanCollide = false
    part.Transparency = 1
    part.Parent = workspace
    
    -- Add particle emitter
    local particle = Instance.new("ParticleEmitter")
    particle.Texture = "rbxassetid://242888855" -- Sparkle texture
    particle.Color = ColorSequence.new(Color3.fromRGB(255, 0, 0))
    particle.Size = NumberSequence.new(1, 3)
    particle.Lifetime = NumberRange.new(2, 5)
    particle.Rate = 50
    particle.SpreadAngle = Vector2.new(180, 180)
    particle.Parent = part
    
    -- Add light
    local light = Instance.new("PointLight")
    light.Color = Color3.fromRGB(255, 0, 0)
    light.Brightness = 2
    light.Range = 50
    light.Parent = part
    
    -- Auto cleanup
    game:GetService("Debris"):AddItem(part, CONFIG.EffectDuration)
end

-- Restore original lighting
local function restoreLighting()
    local tweenInfo = TweenInfo.new(CONFIG.FadeOutTime, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    
    if originalLighting.Brightness then
        TweenService:Create(Lighting, tweenInfo, {Brightness = originalLighting.Brightness}):Play()
    end
    if originalLighting.OutdoorAmbient then
        TweenService:Create(Lighting, tweenInfo, {OutdoorAmbient = originalLighting.OutdoorAmbient}):Play()
    end
    if originalLighting.FogColor then
        TweenService:Create(Lighting, tweenInfo, {
            FogColor = originalLighting.FogColor,
            FogEnd = originalLighting.FogEnd or 100000
        }):Play()
    end
    
    -- Restore skybox after delay
    delay(CONFIG.FadeOutTime, function()
        local sky = Lighting:FindFirstChildOfClass("Sky")
        if sky then
            sky.SkyboxBk = "rbxassetid://6444885997"
            sky.SkyboxDn = "rbxassetid://6444885997"
            sky.SkyboxFt = "rbxassetid://6444885997"
            sky.SkyboxLf = "rbxassetid://6444885997"
            sky.SkyboxRt = "rbxassetid://6444885997"
            sky.SkyboxUp = "rbxassetid://6444885997"
        end
    end)
end

-- Main activation function
local function activateStealthEffects()
    if isActive then return end
    
    isActive = true
    print("üé≠ Stealth effects activated")
    
    -- Apply effects with delays to seem more natural
    applyStealthLighting()
    wait(0.5)
    createGlobalSound()
    wait(0.3)
    createScreenEffects()
    wait(0.2)
    createParticleEffects()
    
    -- Auto deactivate after duration
    delay(CONFIG.EffectDuration, function()
        if isActive then
            deactivateStealthEffects()
        end
    end)
end

-- Deactivation function
local function deactivateStealthEffects()
    if not isActive then return end
    
    isActive = false
    print("‚úÖ Stealth effects deactivated")
    
    restoreLighting()
    
    -- Cleanup connections
    for _, connection in pairs(effectConnections) do
        connection:Disconnect()
    end
    effectConnections = {}
end
